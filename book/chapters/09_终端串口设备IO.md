# 第 9 章 终端、串口与设备 IO

## 9.1 终端与 tty 的本质

Linux 里终端是字符设备，内核用 `tty` 子系统管理。理解其层次对调试串口、PTY 异常行为至关重要：

```
用户程序（bash / vim）
    │  read/write
    ▼
行规程（line discipline，通常 N_TTY）
    │  处理 Ctrl-C/Ctrl-Z/Backspace/回显/换行
    ▼
tty 驱动
    ├── UART 驱动（/dev/ttyS0、/dev/ttyUSB0、/dev/ttyAMA0）
    └── PTY 伪终端（/dev/pts/N）← SSH/tmux/xterm 用这个
```

**行规程（N_TTY）的作用**：
- 输入回显（你打的字显示在屏幕上）
- `Ctrl+C` → 发 SIGINT 给前台进程组
- `Ctrl+Z` → 发 SIGTSTP
- 缓冲一整行，等 `Enter` 再交给程序（规范模式默认行为）

## 9.2 termios —— 控制终端属性

```c
#include <termios.h>
int tcgetattr(int fd, struct termios *termios_p);
int tcsetattr(int fd, int optional_actions, const struct termios *termios_p);
/* optional_actions: TCSANOW | TCSADRAIN | TCSAFLUSH */
```

`struct termios` 核心字段：

| 字段 | 含义 |
| --- | --- |
| `c_iflag` | 输入标志（奇偶处理、软件流控、换行转换） |
| `c_oflag` | 输出标志（换行转换 `\n`→`\r\n`） |
| `c_cflag` | 控制标志（波特率、数据位、停止位、奇偶） |
| `c_lflag` | 本地标志（规范/原始模式、回显、信号字符） |
| `c_cc[]` | 特殊字符（VINTR=Ctrl-C, VEOF=Ctrl-D, VTIME, VMIN） |

### 规范模式 vs 原始模式

**规范模式（Canonical）**：默认，行缓冲，有回显和特殊字符处理，`read` 等到 `\n` 才返回。适合 shell。

**原始模式（Raw）**：无缓冲，无回显，`read` 按字节/超时返回。适合 vim、串口协议解析。

```c
struct termios raw, orig;
tcgetattr(STDIN_FILENO, &orig);   /* 保存原始设置 */
raw = orig;

raw.c_lflag &= ~(ECHO | ICANON | ISIG | IEXTEN);
raw.c_iflag &= ~(IXON | ICRNL | BRKINT | INPCK | ISTRIP);
raw.c_oflag &= ~OPOST;
raw.c_cflag |= CS8;
raw.c_cc[VMIN]  = 0;
raw.c_cc[VTIME] = 1;    /* 100ms 超时 */

tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw);

char c;
read(STDIN_FILENO, &c, 1);   /* 读一个按键，不等回车 */

tcsetattr(STDIN_FILENO, TCSAFLUSH, &orig);   /* 恢复 */
```

## 9.3 串口编程（UART）

### 打开串口

```c
int fd = open("/dev/ttyUSB0", O_RDWR | O_NOCTTY | O_CLOEXEC);
/* O_NOCTTY：防止串口成为进程控制终端 */
```

常见串口设备名：

| 设备名 | 说明 |
| --- | --- |
| `/dev/ttyS0`~`ttyS3` | 主板内置 UART |
| `/dev/ttyUSB0` | USB 转串口（CH340/CP2102/FTDI） |
| `/dev/ttyAMA0` | ARM AMBA UART（树莓派等） |
| `/dev/ttyACM0` | USB CDC-ACM（Arduino 等） |
| `/dev/ttymxc0` | i.MX 系列 UART |

### 配置波特率与通信参数

```c
#include <termios.h>

void serial_config(int fd, int baud)
{
    struct termios tty;
    tcgetattr(fd, &tty);

    speed_t spd = (baud == 115200) ? B115200 :
                  (baud == 9600)   ? B9600   : B115200;
    cfsetispeed(&tty, spd);
    cfsetospeed(&tty, spd);

    /* 8N1 */
    tty.c_cflag &= ~PARENB;   /* 无奇偶校验 */
    tty.c_cflag &= ~CSTOPB;   /* 1 停止位 */
    tty.c_cflag  = (tty.c_cflag & ~CSIZE) | CS8;
    tty.c_cflag &= ~CRTSCTS;  /* 关硬件流控 */
    tty.c_cflag |= CREAD | CLOCAL;

    /* 原始模式 */
    tty.c_lflag &= ~(ICANON | ECHO | ECHOE | ISIG);
    tty.c_iflag &= ~(IXON | IXOFF | IXANY | ICRNL | INLCR);
    tty.c_oflag &= ~OPOST;

    /* 读超时：VTIME * 100ms，VMIN 最少字节数 */
    tty.c_cc[VTIME] = 10;   /* 1 秒超时 */
    tty.c_cc[VMIN]  = 0;

    tcsetattr(fd, TCSANOW, &tty);
    tcflush(fd, TCIOFLUSH);   /* 清空收发缓冲 */
}
```

### VMIN 和 VTIME 的组合语义

| VMIN | VTIME | read 返回时机 |
| --- | --- | --- |
| >0 | 0 | 至少收到 VMIN 字节（永久阻塞等） |
| 0 | >0 | 超时（VTIME×100ms）内有多少取多少，0 也返回 |
| >0 | >0 | 收到第 1 字节后开始计时，超时或满 VMIN 返回 |
| 0 | 0 | 立即返回，有多少取多少 |

### 串口读写示例

```c
serial_config(fd, 115200);

/* 发 AT 命令 */
const char *cmd = "AT\r\n";
write(fd, cmd, strlen(cmd));

/* 读响应（1 秒超时） */
char buf[256];
ssize_t n = read(fd, buf, sizeof(buf) - 1);
if (n > 0) { buf[n] = '\0'; printf("收到: %s\n", buf); }
```

### 串口调试工具

```bash
stty -F /dev/ttyUSB0 115200 cs8 -parenb -cstopb raw
picocom /dev/ttyUSB0 --baud 115200   # 交互终端
cat /dev/ttyUSB0 &                   # 后台读
echo "hello" > /dev/ttyUSB0          # 发送
```

## 9.4 ioctl —— 设备控制

字符设备除了 read/write，驱动还通过 `ioctl` 暴露设备特有控制接口：

```c
#include <sys/ioctl.h>
int ioctl(int fd, unsigned long request, ...);
```

常用 ioctl 示例：

```c
/* 终端窗口大小 */
struct winsize ws;
ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws);
printf("行 %d，列 %d\n", ws.ws_row, ws.ws_col);

/* 网卡 IP 地址 */
struct ifreq ifr = {};
strncpy(ifr.ifr_name, "eth0", IFNAMSIZ);
int s = socket(AF_INET, SOCK_DGRAM, 0);
ioctl(s, SIOCGIFADDR, &ifr);
printf("eth0: %s\n", inet_ntoa(
    ((struct sockaddr_in *)&ifr.ifr_addr)->sin_addr));
close(s);
```

## 9.5 GPIO 编程（嵌入式）

Linux 提供两套 GPIO 接口：

### 旧接口：sysfs（仍广泛使用）

```bash
echo 17 > /sys/class/gpio/export
echo out > /sys/class/gpio/gpio17/direction
echo 1   > /sys/class/gpio/gpio17/value    # 高电平
echo 0   > /sys/class/gpio/gpio17/value    # 低电平
```

C 代码：

```c
void gpio_set(int gpio_num, int value)
{
    char path[64];
    snprintf(path, sizeof(path), "/sys/class/gpio/gpio%d/value", gpio_num);
    int fd = open(path, O_WRONLY);
    write(fd, value ? "1" : "0", 1);
    close(fd);
}
```

### 新接口：GPIO chardev + libgpiod（Linux 4.8+，推荐）

```c
#include <gpiod.h>   /* 链接 -lgpiod */

struct gpiod_chip *chip = gpiod_chip_open("/dev/gpiochip0");
struct gpiod_line *line = gpiod_chip_get_line(chip, 17);
gpiod_line_request_output(line, "myapp", 0);
gpiod_line_set_value(line, 1);   /* 拉高 */
sleep(1);
gpiod_line_set_value(line, 0);   /* 拉低 */
gpiod_line_release(line);
gpiod_chip_close(chip);
```

**为什么用新接口**：sysfs GPIO 在多进程竞争时没有保护机制，新 chardev 接口在内核层面有独占锁，并且支持中断/边沿事件通知（`gpiod_line_event_wait`）。

## 9.6 I²C 编程

```c
#include <linux/i2c-dev.h>
#include <sys/ioctl.h>

int fd = open("/dev/i2c-1", O_RDWR);
ioctl(fd, I2C_SLAVE, 0x48);    /* 7 位从设备地址 */

/* 写寄存器 */
uint8_t tx[2] = { 0x01, 0xC3 };   /* [0]=寄存器地址，[1]=数据 */
write(fd, tx, 2);

/* 读寄存器：先写地址，再读 */
uint8_t reg = 0x00;
write(fd, &reg, 1);
uint8_t rx[2];
read(fd, rx, 2);
int16_t raw = (rx[0] << 8) | rx[1];
printf("温度: %.2f °C\n", raw / 256.0);

close(fd);
```

### I²C 调用链

```
用户态 write(fd, tx, 2)
  └─ i2c-dev 驱动的 .write()
       └─ i2c_master_send()
            └─ i2c 适配器驱动（i2c-omap / i2c-designware 等）
                 └─ 产生 START + 地址 + 数据 + STOP 信号（硬件 FIFO）
```

## 9.7 SPI 编程

```c
#include <linux/spi/spidev.h>

int fd = open("/dev/spidev0.0", O_RDWR);

uint8_t mode = SPI_MODE_0; uint32_t speed = 1000000; uint8_t bits = 8;
ioctl(fd, SPI_IOC_WR_MODE,          &mode);
ioctl(fd, SPI_IOC_WR_MAX_SPEED_HZ,  &speed);
ioctl(fd, SPI_IOC_WR_BITS_PER_WORD, &bits);

/* 全双工传输 */
uint8_t tx[] = { 0x9F, 0, 0, 0 };
uint8_t rx[4] = {};
struct spi_ioc_transfer tr = {
    .tx_buf = (uintptr_t)tx,
    .rx_buf = (uintptr_t)rx,
    .len    = 4,
    .speed_hz      = speed,
    .bits_per_word = bits,
};
ioctl(fd, SPI_IOC_MESSAGE(1), &tr);
printf("JEDEC ID: %02X %02X %02X\n", rx[1], rx[2], rx[3]);
close(fd);
```

## 9.8 /dev/mem 直接访问物理寄存器

```c
/* 仅用于调试/原型，需要 root */
int fd = open("/dev/mem", O_RDWR | O_SYNC);
volatile uint32_t *reg = mmap(NULL, 0x1000,
    PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0xFEC00000);
uint32_t val = reg[0];   /* 读 */
reg[1] = 0x01;           /* 写 */
munmap((void *)reg, 0x1000);
close(fd);
/* O_SYNC 阻止 CPU 写合并，保证寄存器操作时序正确 */
```

## 9.9 PTY 伪终端 —— 实现 SSH/tmux/expect

PTY 是一对字符设备（master/slave），SSH 服务端、tmux、expect 都用它让程序"以为自己在和真实终端交互"：

```c
#define _GNU_SOURCE
#include <pty.h>   /* 链接 -lutil */

int master_fd;
pid_t pid = forkpty(&master_fd, NULL, NULL, NULL);
if (pid == 0) {
    execlp("bash", "bash", NULL);   /* 子进程运行在 PTY slave 上 */
    _exit(1);
}
/* 父进程通过 master_fd 和 bash 交互 */
write(master_fd, "ls -l\n", 6);
char buf[4096];
ssize_t n = read(master_fd, buf, sizeof(buf));
write(STDOUT_FILENO, buf, n);
```

## 9.10 易错点汇总

- **串口没设 O_NOCTTY**：串口成为控制终端，Ctrl-C 发 SIGINT 给整个进程组
- **VMIN=1 + VTIME=0**：没数据时 read 永久阻塞，适合全量接收；想要超时需设 VTIME>0
- **没 tcflush 就读**：缓冲区有旧数据，读到期望外的内容
- **ioctl 命令码对象不对**：SIOCGIFADDR 要发给网络 socket，不能发给串口 fd
- **sysfs GPIO 没设方向就写 value**：默认输入，写无效且静默失败
- **I²C 从设备地址重叠**：`I2C_SLAVE` 是进程级的，多进程访问同地址设备需要用 `I2C_SLAVE_FORCE`
- **SPI tx/rx 长度不一致**：`spi_ioc_transfer.len` 是发送和接收共用，必须等长
